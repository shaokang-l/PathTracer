# ======================================================================== #
# my-pt - minimal interactive OWL path-tracing scaffold.
#
# How to use this file:
#   - It assumes the OWL repo has been added with add_subdirectory(),
#     so targets   owl::owl   owl_viewer   and the embed_ptx() function
#     are available.
#   - Inside the OWL repo (default when this folder lives at
#     <owl_repo>/my-pt/) you can just do:
#
#         add_subdirectory(my-pt)
#
#     somewhere in the top-level CMakeLists.txt.
#
#   - In your own project (the recommended setup), put OWL as a git
#     submodule, then in your project's top-level CMakeLists.txt:
#
#         add_subdirectory(third_party/owl EXCLUDE_FROM_ALL)
#         add_subdirectory(my-pt)
#
#     See my-pt/README.md for full instructions.
# ======================================================================== #

cmake_minimum_required(VERSION 3.18)

if(NOT TARGET owl::owl)
  message(FATAL_ERROR
    "my-pt requires the OWL 'owl::owl' target. "
    "Did you add_subdirectory(<path-to-owl>) before adding this folder?")
endif()

if(NOT TARGET owl_viewer)
  message(FATAL_ERROR
    "my-pt requires the 'owl_viewer' target (interactive window). "
    "Make sure GLFW / OpenGL are available in your OWL build.")
endif()

# Force C++17 in this subdir: the parent PathTracer project uses C++20,
# but CUDA 11.x / nvcc 11.8 does not support C++20 for device code, and
# the target generated by embed_ptx inherits the global CMAKE_CXX_STANDARD.
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CUDA_STANDARD 17)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

embed_ptx(
  OUTPUT_TARGET
    mypt-ptx
  PTX_LINK_LIBRARIES
    owl::owl
  SOURCES
    src/deviceCode.cu
)

target_include_directories(mypt-ptx
  PRIVATE
    ${CMAKE_SOURCE_DIR}/common/include
)

add_executable(mypt
  src/main.cpp
  src/Denoiser.cpp
  src/Renderer.cpp
  src/Scene.cpp
  src/Viewer.cpp
  src/postprocess.cu
)

set_target_properties(mypt PROPERTIES
  CXX_STANDARD 17
  CXX_STANDARD_REQUIRED ON
  CUDA_STANDARD 17
  CUDA_STANDARD_REQUIRED ON
  CUDA_SEPARABLE_COMPILATION ON
)

target_include_directories(mypt
  PRIVATE
    src
    ${CMAKE_SOURCE_DIR}/common/include
    ${CMAKE_SOURCE_DIR}/cpu/include
)

target_compile_definitions(mypt
  PRIVATE
    PATHTRACER_ASSET_DIR="${CMAKE_SOURCE_DIR}/assets"
)

target_link_libraries(mypt
  PRIVATE
    mypt-ptx
    owl::owl
    owl_viewer
)

# ------------------------------------------------------------------
# On Windows OWL builds as a shared library (owl.dll). Because OWL
# lives under third_party/owl/owl/ in the build tree, Windows can't
# find the DLL when launching mypt.exe from build/gpu/. Copy every
# runtime DLL the executable transitively depends on next to the
# binary after linking. (Silently no-ops when owl is static, e.g.
# on Linux, because TARGET_RUNTIME_DLLS resolves to an empty list.)
# ------------------------------------------------------------------
if(WIN32)
  add_custom_command(TARGET mypt POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
      $<TARGET_RUNTIME_DLLS:mypt>
      $<TARGET_FILE_DIR:mypt>
    COMMAND_EXPAND_LISTS
    VERBATIM
  )
endif()
